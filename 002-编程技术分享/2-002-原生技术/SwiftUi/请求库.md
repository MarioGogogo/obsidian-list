# Alamofire

```swift
import Alamofire // 需要先引入 Alamofire 库

// 假设已经定义了 Post 结构体， Alamofire 可以直接解析
// @State private var posts: [Post] = [] // 在 SwiftUI 视图里

// 在某个函数里
func fetchPostsWithAlamofire() {
    AF.request("https://jsonplaceholder.typicode.com/posts").responseDecodable(of: [Post].self) { response in
        switch response.result {
            case .success(let posts):
                // 成功了！posts 就是已经帮你解析好的 [Post] 数组
                // 在 SwiftUI 里，你需要更新 @State 变量：
                // self.posts = posts
                print("Success: \(posts.count) posts")
            case .failure(let error):
                print("Error: \(error)")
        }
    }
}

```
